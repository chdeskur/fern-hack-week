## Code Blocks

### JavaScript with Syntax Highlighting
```javascript
// Simple function with template literals
const processData = (data, options = {}) => {
  const { 
    transform = (x) => x, 
    filter = () => true,
    sort = (a, b) => a - b 
  } = options;
  
  return data
    .filter(filter)
    .map(transform)
    .sort(sort);
};

// Usage with simple props
const result = processData(users, {
  transform: (user) => ({
    ...user,
    fullName: `${user.firstName} ${user.lastName}`
  }),
  filter: (user) => user.active,
  sort: (a, b) => a.name.localeCompare(b.name)
});
```

### TypeScript with Generics
```typescript
interface ApiResponse<T> {
  data: T;
  status: 'success' | 'error';
  message?: string;
}

class ApiClient<T> {
  constructor(private baseUrl: string) {}
  
  async fetch<R = T>(endpoint: string): Promise<ApiResponse<R>> {
    const response = await fetch(`${this.baseUrl}${endpoint}`);
    return response.json();
  }
}

// Usage
const userClient = new ApiClient<User>('/api/users');
const users = await userClient.fetch<User[]>('/list');
```
